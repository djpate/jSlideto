Default example

$("#title3").slideto();

No highlight

$("#title3").slideto({
	highlight: false
});

Red & long highlight

$("#title3").slideto({
	highlight_color: 'red',
	highlight_duration : 5000
});

Offset

$("#title3").slideto({
	offset: 60
});

All options

$("#title3").slideto({
	highlight_color: 'blue',
	highlight_duration : 500,
	slide_duration : 'fast'
});
